339template<int MAX_LAYER_COUNT>
340class DDFadingLayers {
341 public:
342 DDFadingLayers() {
343 this->layerCount = 0;
344 this->nextUseLayerIdx = 0;
345 }
346 void initAddLayer(DDLayer* layer) {
347 if (layerCount < MAX_LAYER_COUNT) {
348 layers[layerCount++] = layer;
349 }
350 }
351 void clear() {
352 for (int i = 0; i < layerCount; i++) {
353 layers[i]->clear();
354 }
355 nextUseLayerIdx = 0;
356 }
357 public:
358 DDLayer* useLayer() {
359 int layerIdx = nextUseLayerIdx;
360 nextUseLayerIdx = (nextUseLayerIdx + 1) % layerCount;
361 for (int i = 0; i < this->layerCount; i++) {
362 int lidx = (layerIdx + i) % layerCount;
363 DDLayer* layer = layers[lidx];
364 if (i == 0) {
365 layer->opacity(100);
366 layer->clear();
367 } else {
368 int opacity = i * (100.0 / layerCount);
369 layer->opacity(opacity);
370 }
371 }
372 return layers[layerIdx];
373 }
374 private:
375 DDLayer* layers[MAX_LAYER_COUNT];
376 int layerCount;
377 int nextUseLayerIdx;
378};
