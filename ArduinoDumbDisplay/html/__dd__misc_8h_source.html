360template<int MAX_LAYER_COUNT>
361class DDFadingLayers {
362 public:
363 DDFadingLayers() {
364 this->layerCount = 0;
365 this->nextUseLayerIdx = 0;
366 }
367 void initAddLayer(DDLayer* layer) {
368 if (layerCount < MAX_LAYER_COUNT) {
369 layers[layerCount++] = layer;
370 }
371 }
372 void clear() {
373 for (int i = 0; i < layerCount; i++) {
374 layers[i]->clear();
375 }
376 nextUseLayerIdx = 0;
377 }
378 public:
379 DDLayer* useLayer() {
380 int layerIdx = nextUseLayerIdx;
381 nextUseLayerIdx = (nextUseLayerIdx + 1) % layerCount;
382 for (int i = 0; i < this->layerCount; i++) {
383 int lidx = (layerIdx + i) % layerCount;
384 DDLayer* layer = layers[lidx];
385 if (i == 0) {
386 layer->opacity(100);
387 layer->clear();
388 } else {
389 int opacity = i * (100.0 / layerCount);
390 layer->opacity(opacity);
391 }
392 }
393 return layers[layerIdx];
394 }
395 private:
396 DDLayer* layers[MAX_LAYER_COUNT];
397 int layerCount;
398 int nextUseLayerIdx;
399};
